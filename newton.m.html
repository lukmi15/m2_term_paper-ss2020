1 % Newton-Verfahren für nichtlineare Gleichungssysteme
 2 %
 3 %   [x, fval, exitflag] = newton(f, x0, tol, nmax)
 4 %
 5 % Beim Newton-Verfahren soll mindestens ein Schritt durchgeführt werden
 6 % und die Iteration abgebrochen werden, falls für die k-te Iterierte x_k
 7 %
 8 %   || f(x_k) || + || x_k - x_(k-1) || < tol
 9 %
10 % gilt. Hier ist ||.|| die (euklidsche) Länge eines Vektors. Rückgabewert
11 % ist dann x_k ansonsten x_nmax.
12 %
13 % Eingabe:
14 %   f        Funktionshandle auf Funktionshandle [y,dy] = f(x)
15 %               Eingabe:
16 %                   x   Spaltenvektor der Unbekannten
17 %               Ausgabe:
18 %                   y   Spaltenvektor des Funktionswertes an x
19 %                   dy  Jacobi-Matrix an x
20 %   x0       Spaltenvektor des Startwerts x_0
21 %   tol  Toleranz
22 %   nmax     maximale Anzahl an Iterationen bis Abbruch
23 %
24 % Ausgabe
25 %   x        Spaltenvektor der näherungsweisen Nullstelle
26 %   fval     Spaltenvektor des Funktionswertes an x
27 %   exitflag Rückgabewert 1 falls Iteration Toleranzbedingung erfüllt hat
28 %            ansonsten 0
29 %
30 
31 function [x, fval, exitflag] = newton(f, x0, tol, nmax)
32     exitflag = 0; %Not needed
33     x = x0;
34     k = 1;
35     x(:, k)
36     [fval, J] = f(x(:, k));
37     k++;
38     x(:, k) = x(:, k - 1) - J \ fval;
39     while k <= nmax && norm(fval) + norm(x(:, k) - x(:, k - 1)) >= tol
40         [fval, J] = f(x(:, k));
41         exit
42         k++;
43         x(:, k) = x(:, k - 1) - J \ fval;
44     end
45     x = x(:, end);
46     [fval, ~] = f(x);
47 end
